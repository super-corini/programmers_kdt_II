from django.urls import path
from . import views


urlpatterns = [
    path('index/',  views.index),
    path('intro/', views.intro),
    path('coffee/', views.coffee_view),
    path('burger/', views.burger),
    path('burger_add', views.burger_add),
    path('burger_update/<int:burger_id>/', views.burger_update, name='burger_update'),
    path('burger_delete<int:burger_id>/', views.burger_delete, name='burger_delete'),
]

###########################views.py################################################

from django.shortcuts import render, HttpResponse, redirect, get_object_or_404
from .models import Coffee, Burger
from .forms import CoffeeForm, BurgerForm

# Create your views here.
def index(request):
    # return HttpResponse("<H1>Hello World")
    nums = [1, 2, 3, 4, 5]
    return render(request, 'index.html', {'my_n': nums})

def intro(request):
    return render(request, 'intro.html')

def coffee_view(request):
    coffee_all = Coffee.objects.all()
    # 만약 request가 POST라면:
        #POST를 바탕으로Forem을 완성하고 유효하면 저장
    if request.method == 'POST':
        form = CoffeeForm(request.POST)
        if form.is_valid():
            form.save()

    form = CoffeeForm()
    return render(request, 'coffee.html', { "coffee_list" : coffee_all, "coffee_form": form })

def burger(request):
    burger = Burger.objects.all()
    form = BurgerForm()
    burger_info = {
        'burger_list': burger,
        'burger_form': form,
    }
    return render(request, 'burger.html', burger_info)

def burger_add(request):
    form = BurgerForm()
    if request.method == 'POST':
        form = BurgerForm(request.POST)
        if form.is_valid():
            form.save()
        return redirect('/hompage/burger')
    return render(request, 'burger_form.html', {'burger_form': form})
    
def burger_update(request, burger_id):
    burger = get_object_or_404(Burger, pk=burger_id)
    form = BurgerForm(instance=burger)
    if request.method == 'POST':
        form = BurgerForm(request.POST, instance=burger)
        if form.is_valid():
            bur = form.save(commit=False)
            bur.save()
            return redirect('/hompage/burger')
    return render(request, 'burger_form.html', {'burger_form': form})

def burger_delete(request, burger_id):
    burger = get_object_or_404(Burger, pk=burger_id)
    burger.delete()
    return redirect('/hompage/burger')

###########################models.py####################################
from django.db import models

# Create your models here.
class Coffee(models.Model):
    def __str__(self):
        return self.name
    name = models.CharField(default="", max_length=30)
    price = models.IntegerField(default=0)
    is_ice = models.BooleanField(default=False)

class Burger(models.Model):
    name = models.CharField(max_length=50)
    price = models.IntegerField()
    is_set = models.BooleanField()

###########################forms.py####################################
from django import forms
from .models import Coffee, Burger


class CoffeeForm(forms.ModelForm):#MoodelForm을 상속받는 CoffeeForm 생성
    class Meta:
        model = Coffee
        fields = ('__all__')#'name', 'price', 'is_ice'

class BurgerForm(forms.ModelForm):
    class Meta:
        model = Burger
        fields = ('__all__')
